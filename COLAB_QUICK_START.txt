class LossScaler:
    def __init__(self, scale):
        self.scale = scale
        self.growth_factor = 2.0
        self.backoff_factor = 0.5
    
    @staticmethod
    def for_transformer(num_layers):
        return LossScaler(256.0 * (num_layers ** 0.5))
    
    def scale_loss(self, loss):
        return loss * self.scale
    
    def update(self, grads):
        has_overflow = any(g != g or abs(g) > 1e10 for g in grads)
        if has_overflow:
            self.scale *= self.backoff_factor
            return False
        return True
    
    def current_scale(self):
        return self.scale

print("Testing LossScaler...")
scaler = LossScaler.for_transformer(12)
print(f"✅ Scale for 12-layer model: {scaler.current_scale():.1f}")

loss = 0.5
scaled = scaler.scale_loss(loss)
print(f"✅ Loss scaling: {loss} → {scaled}")

grads = [1e20, 2e20]
valid = scaler.update(grads)
print(f"✅ Overflow detection: overflow={not valid}, new scale={scaler.current_scale():.1f}")

class GradAccumulator:
    def __init__(self, num_params, total_steps):
        self.accum = [0.0] * num_params
        self.count = 0
        self.total_steps = total_steps
    
    def accumulate_fp32(self, grads):
        self.count += 1
        for i, g in enumerate(grads):
            self.accum[i] += g
    
    def finalize_averaged(self):
        return [g / self.count for g in self.accum]
    
    def is_complete(self):
        return self.count >= self.total_steps
    
    def reset(self):
        self.accum = [0.0] * len(self.accum)
        self.count = 0

print("Testing GradAccumulator...")
accum = GradAccumulator(5000, 4)
print(f"✅ Created: {accum.total_steps} steps, {len(accum.accum)} params")

for step in range(4):
    grads = [0.5] * 5000
    accum.accumulate_fp32(grads)

avg = accum.finalize_averaged()
print(f"✅ Accumulated and averaged: {avg[0]:.2f}")

accum.reset()
print(f"✅ Reset: {accum.count}/{accum.total_steps}")

Expected:
✅ Created: 4 steps, 5000 params
✅ Accumulated and averaged: 0.50
✅ Reset: 0/4

import random

print("Simulating 10 training steps...\n")

scaler = LossScaler.for_transformer(12)
accum = GradAccumulator(5000, 4)

for step in range(10):
    loss = 2.0 - (step * 0.1) + random.uniform(-0.05, 0.05)
    scaled_loss = scaler.scale_loss(loss)
    grads = [scaled_loss / 100] * 5000
    accum.accumulate_fp32(grads)
    valid = scaler.update(grads)
    
    status = "✅" if valid else "⚠️"
    print(f"Step {step+1:2d}: loss={loss:.3f}, scale={scaler.current_scale():.0f}, "
          f"accum={accum.count}/4 {status}", end="")
    
    if accum.is_complete():
        print(" → UPDATE!")
        accum.reset()
    else:
        print()

print("\n✅ Simulation complete!")
