```rust
//! Delta compilation manager for multi-layer ANE training.
//!
//! DeltaCompiler owns multiple ANEExecutor instances (one per program),
//! tracks the ANE compile budget, and provides efficient weight reloading
//! for the training loop.

use crate::ane::runtime::CompileBudgetMonitor;
use crate::wrapper::{ANECompiler, ANEExecutor};
use crate::{Error, Result};
use std::time::Instant;

/// A single compiled ANE program with its weight file metadata.
struct CompiledLayer {
    /// Owned executor (owns the ANE kernel handle)
    executor: ANEExecutor,
    /// Weight file paths for reload
    weight_names: Vec<String>,
    /// Number of inputs
    num_inputs: usize,
    /// Number of outputs
    num_outputs: usize,
}

/// Multi-layer delta compilation manager.
///
/// Compiles ANE programs once, then uses delta compilation (weight reload)
/// for fast weight updates during training.
pub struct DeltaCompiler {
    /// Compiled layers (forward + backward programs)
    layers: Vec<CompiledLayer>,
    /// Compile budget monitor
    monitor: CompileBudgetMonitor,
    /// Total compiles used at construction time
    initial_compile_count: i32,
}

impl DeltaCompiler {
    /// Create a new DeltaCompiler.
    pub fn new() -> Self { ... }

    /// Compile and add a program (forward or backward).
    /// weight_names: e.g. ["@model_path/weights/w1.bin", ...]
    /// Returns layer index.
    pub fn add_program(
        &mut self,
        mil_text: &str,
        weight_names: &[&str],
        weight_datas: &[&[u8]],
        input_sizes: &[usize],
        output_sizes: &[usize],
    ) -> Result<usize> { ... }

    /// Reload weights for a specific layer.
    pub fn reload_layer(&mut self, layer_idx: usize, weight_updates: &[(&str, &[u8])]) -> Result<Duration> { ... }

    /// Reload weights for ALL layers. Returns total time.
    /// weight_updates[layer_idx] = &[("weight_name.bin", &data), ...]
    pub fn reload_all(&mut self, weight_updates: &[Vec<(&str, &[u8])>]) -> Result<Duration> { ... }

    /// Get executor for a layer (for write_input/eval/read_output).
    pub fn executor(&self, layer_idx: usize) -> Result<&ANEExecutor> { ... }

    /// Get mutable executor for a layer.
    pub fn executor_mut(&mut self, layer_idx: usize) -> Result<&mut ANEExecutor> { ... }

    /// Number of compiled programs.
    pub fn num_programs(&self) -> usize { ... }

    /// Current ANE compile count (process-wide).
    pub fn compile_count(&self) -> i32 { ... }

    /// Remaining compile budget.
    pub fn remaining_budget(&self) -> i32 { ... }

    /// Check if approaching compile limit (emits warning).
    /// Returns true if in warning zone.
    pub fn check_budget_warning(&self) -> bool { ... }

    /// Check if budget exhausted.
    pub fn is_budget_exhausted(&self) -> bool { ... }
}
```

Implementation notes:
- add_program() should call ANECompiler::compile_multi() and store the executor
- After adding all programs, check_budget_warning() should be called
- reload_layer() measures timing with Instant, calls executor.reload_weights()
- reload_all() iterates all layers, accumulates time
- check_budget_warning() uses CompileBudgetMonitor::is_warning_zone() and prints to stderr
- All error cases properly mapped to crate::Error
